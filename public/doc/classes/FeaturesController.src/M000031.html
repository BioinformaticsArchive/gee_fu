  def lookup(query, id)
    response = new_response
    rows = Feature.find(:all, :conditions => ["experiment_id = ? and features.group like ?", id, "%" + query + "%"]).collect! {|f| f.to_lookup}
    response[:count] = rows.length
    response[:rows] = rows
    return response
  end
